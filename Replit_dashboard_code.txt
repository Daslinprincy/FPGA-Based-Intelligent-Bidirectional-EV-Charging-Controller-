Code 1: Frontend (HTML, CSS & JavaScript)

This code develops a web-based EV Battery Monitoring Dashboard that displays real-time battery parameters such as State of Charge (SOC), voltage, current, temperature, and cycle count. It provides a graphical battery indicator and allows users to control charging, discharging, and stop operations. The dashboard continuously fetches battery data from the backend and updates the display dynamically.

    <script>
      let lastSoc = null,
        dischargeStart = null,
        dischargeTarget = null,
        discharging = false;

      function batteryColor(soc) {
        return soc > 50 ? "#00ff88" : soc > 20 ? "#ffaa00" : "#ff4444";
      }

      function updateBattery(soc) {
        const fill = document.getElementById("batt-fill");
        fill.setAttribute(
          "width",
          Math.min(102, Math.max(0, (soc / 100) * 102)),
        );
        fill.style.fill = batteryColor(soc);
        document.getElementById("batt-text").textContent = soc.toFixed(1) + "%";
      }

      function updateDisplay(d) {
        const soc = parseFloat(d.soc),
          voltage = parseFloat(d.voltage),
          current = parseFloat(d.current),
          temp = parseFloat(d.temp);
        if (!isNaN(soc)) {
          lastSoc = soc;
          updateBattery(soc);
          const pct = soc.toFixed(1);
          document.getElementById("charge-soc").innerHTML =
            pct + "<span> %</span>";
          document.getElementById("discharge-soc").innerHTML =
            pct + "<span> %</span>";
        }
        if (!isNaN(voltage))
          document.getElementById("voltage").textContent =
            voltage.toFixed(1) + "V";
        if (!isNaN(temp))
          document.getElementById("temp").textContent = temp.toFixed(1) + "°C";
        if (!isNaN(current)) {
          const el = document.getElementById("current");
          if (current > 0.05) {
            el.textContent = "+" + current.toFixed(1) + "A";
            el.className = "val positive";
          } else if (current < -0.05) {
            el.textContent = current.toFixed(1) + "A";
            el.className = "val negative";
          } else {
            el.textContent = "0.0A";
            el.className = "val";
          }
        }
        if (d.cycles != null)
          document.getElementById("cycles").textContent = Math.round(d.cycles);
        const connected = d.source === "fpga";
        document.getElementById("conn-dot").className =
          "conn-dot" + (connected ? " connected" : "");
        document.getElementById("conn-label").textContent = connected
          ? "FPGA CONNECTED"
          : "FPGA IDLE";
        const badge = document.getElementById("mode-badge");
        const btnC = document.getElementById("btn-charge");
        const btnD = document.getElementById("btn-discharge");
        if (!isNaN(current) && current > 0.05) {
          badge.textContent = "CHARGING";
          badge.className = "mode-badge charging";
          btnC.className = "btn charge-active";
          if (!discharging) btnD.className = "btn";
        } else if (!isNaN(current) && current < -0.05) {
          badge.textContent = "DISCHARGING";
          badge.className = "mode-badge discharging";
          btnD.className = "btn discharge-active";
          btnC.className = "btn";
        } else {
          badge.textContent = "IDLE";
          badge.className = "mode-badge";
          if (!discharging) {
            btnC.className = "btn";
            btnD.className = "btn";
          }
        }

        if (
          discharging &&
          dischargeStart !== null &&
          dischargeTarget !== null &&
          !isNaN(soc)
        ) {
          const discharged = dischargeStart - soc;
          const remaining = Math.max(0, dischargeTarget - discharged);
          const prog = document.getElementById("discharge-progress");
          prog.className = "panel-progress active";
          prog.textContent = `Discharged ${discharged.toFixed(1)}% of ${dischargeTarget}% — ${remaining.toFixed(1)}% remaining`;
          if (discharged >= dischargeTarget) {
            sendCommand("STOP", 0);
            discharging = false;
            dischargeStart = null;
            dischargeTarget = null;
            prog.className = "panel-progress";
            prog.textContent = "Discharge complete ✓";
            btnD.className = "btn";
          }
        }
      }

      async function fetchData() {
        try {
          const res = await fetch("/data");
          if (!res.ok) return;
          updateDisplay(await res.json());
        } catch (_) {}
      }

      async function sendCommand(command, delta = 0) {
        try {
          await fetch("/api/command", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ command, delta }),
          });
        } catch (_) {}
      }

      document.getElementById("btn-charge").addEventListener("click", () => {
        discharging = false;
        dischargeStart = null;
        dischargeTarget = null;
        document.getElementById("discharge-progress").textContent = "";
        document.getElementById("discharge-progress").className =
          "panel-progress";
        sendCommand("CHARGE_START", 0);
      });

      document.getElementById("btn-discharge").addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("discharge-amt").value);
        const prog = document.getElementById("discharge-progress");
        if (isNaN(amt) || amt <= 0) {
          prog.textContent = "Enter a valid % to discharge.";
          prog.className = "panel-progress active";
          return;
        }
        if (lastSoc !== null && amt >= lastSoc) {
          prog.textContent = "Target exceeds current SOC.";
          prog.className = "panel-progress active";
          return;
        }
        dischargeStart = lastSoc;
        dischargeTarget = amt;
        discharging = true;
        prog.textContent = `Starting discharge — target: -${amt}%`;
        prog.className = "panel-progress active";
        sendCommand("DISCHARGE_START", amt);
      });

      document.getElementById("btn-stop").addEventListener("click", () => {
        discharging = false;
        dischargeStart = null;
        dischargeTarget = null;
        document.getElementById("discharge-progress").textContent = "";
        document.getElementById("discharge-progress").className =
          "panel-progress";
        sendCommand("STOP", 0);
      });

      fetchData();
      setInterval(fetchData, 1000);
    </script>

Code 2: Backend (Express.js & TypeScript)

This code implements a backend server that acts as an interface between the FPGA/BMS hardware and the web dashboard. It receives battery data from the FPGA, stores the latest battery status, and provides APIs for monitoring and control. The server also manages charging, discharging, and stop commands sent from the dashboard and makes them available for the FPGA to execute.

import { Router, type IRouter, type Request, type Response } from "express";
import healthRouter from "./health";

const router: IRouter = Router();
router.use(healthRouter);

export let bmsData: Record<string, unknown> = {
  soc: 85.0,
  voltage: 59.4,
  current: 0.0,
  temp: 34.0,
  source: "waiting",
};

export let pendingCmd: Record<string, unknown> = { command: "none" };

router.post("/bms-data", (req: Request, res: Response) => {
  const d = req.body || {};
  bmsData = { ...bmsData, ...d, source: "fpga" };
  res.json({ ok: true });
});

router.post("/command", (req: Request, res: Response) => {
  pendingCmd = req.body || { command: "none" };
  res.json({ ok: true });
});

router.get("/command", (req: Request, res: Response) => {
  const cmd = pendingCmd;
  pendingCmd = { command: "none" };
  res.json(cmd);
});

export default router;
